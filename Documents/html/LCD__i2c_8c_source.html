319 void putHexString(unsigned char c, unsigned char r, unsigned char hexval) {
320  LCDsetCursor(0 + (5 * c), 2 + (2 * r));
321  LCD_Write_Str("0x");
322  LCDsetCursor(2 + (5 * c), 2 + (2 * r));
323  LCD_Write_Char((char) hex[(hexval >> 4) & 0x0f]);
324  LCDsetCursor(3 + (5 * c), 2 + (2 * r));
325  LCD_Write_Char((char) hex[hexval & 0x0f]);
