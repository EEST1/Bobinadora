90 //
91 // Code was written with the following assumptions as to PCF8574 -> Parallel 4bit convertor interconnections
92 // controlling a 20 by 4 LCD display. Assumes A0...A2 on PCF8574 are all pulled high. Giving address of 0x4E or 0b01001110 (0x27)
93 // (the last bit [b0] is I2C R/nW bit)
94 //
95 // Pin out for LCD display (16 pins)
96 // ---------------------------------
97 // 1 - Gnd
98 // 2 - Vcc
99 // 3 - VContrast
100 // 4 - RS - P0 - Pin 4 PCF8574
101 // 5 - RW - P1 - Pin 5 PCF8574
102 // 6 - En - P2 - Pin 6 PCF8574
103 // 7 - D0 - Don't Care
104 // 8 - D1 - Don't Care
105 // 9 - D2 - Don't Care
106 // 10 - D3 - Don't Care
107 // 11 - D4 - P4 - Pin 9 PCF8574
108 // 12 - D6 - P5 - Pin 10 PCF8574
109 // 13 - D6 - P6 - Pin 11 PCF8574
110 // 14 - D7 - P7 - Pin 12 PCF8574
111 // 15 - Anode LED
112 // 16 - Cathode LED
113 //
114 // PCF8574 register and pin mapping
115 // Bit 0 - RS - P0 - Pin 4 PCF8574
116 // Bit 1 - RW - P1 - Pin 5 PCF8574
117 // Bit 2 - En - P2 - Pin 6 PCF8574
118 // Bit 3 - Led - P3 - Pin 7 PCF8574 (Active High, Led turned on)
119 // Bit 4 - D4 - P4 - Pin 9 PCF8574
120 // Bit 5 - D5 - P5 - Pin 10 PCF8574
121 // Bit 6 - D6 - P6 - Pin 11 PCF8574
122 // Bit 7 - D7 - P7 - Pin 12 PCF8574
123 //

179 void LCD_Init(void){
180  uint8_t culo;
181  //_backlightval &= ~Bl; // Off at start up
182  _backlightval |= Bl; // On at start up
183  _numlines = LCD_MAX_ROWS;
184 
185  // Ensure supply rails are up before config sequence
186  delayInaccurateUs(50000);
187 
188  // Set all control and data lines low. D4 - D7, En (High=1), Rw (Low = 0 or Write), Rs (Control/Instruction) (Low = 0 or Control)
189  // I2C_Write_Byte_Single_Reg(LCD_PCF8574_ADDR, LCD_INIT); // Backlight off (Bit 3 = 0)
190  culo = LCD_INIT;
191 
192  i2cWrite(I2C0,0x27,&culo,1,TRUE);
193  delayInaccurateUs(100);
194 
195  // Sequence to put the LCD into 4 bit mode this is according to the hitachi HD44780 datasheet page 109
196 
197  // we start in 8bit mode
198  LCDwrite4bits(LCD_8BIT_INIT);
199  delayInaccurateUs(4500); // wait more than 4.1ms
200 
201  // second write
202  LCDwrite4bits(LCD_8BIT_INIT);
203  delayInaccurateUs(150); // wait > 100us
204 
205  // third write
206  LCDwrite4bits(LCD_8BIT_INIT);
207  delayInaccurateUs(150);
208 
209  // now set to 4-bit interface
210  LCDwrite4bits(LCD_4BIT_INIT);
211  delayInaccurateUs(150);
212 
213  // set # lines, font size, etc.
214  _functionset = LCD_INTF4BITS | LCD_TWO_LINES | LCD_FONT_5_7;
215  LCDcommandWrite(LCD_FUNCTION_SET | _functionset);
216  //DelayMicroseconds(150);
217 
218  _displayfunction = LCD_DISPLAY_OFF | LCD_CURSOR_OFF | LCD_BLINKING_OFF;
219  LCDdisplayOff();
220 
221  // turn the display on with no cursor or blinking default
222  LCDdisplayOn();
223 
224  // set the entry mode
225  _entrymodeset = LCD_INCREMENT | LCD_SHIFT_OFF; // Initialize to default text direction (for roman languages)
226  LCDcommandWrite(LCD_ENTRY_MODE_SET | _entrymodeset);
227 
228  // Display Function set
229  // _displayfunction = LCD_DISPLAY_ON | LCD_CURSOR_OFF | LCD_BLINKING_OFF;
230  LCDcommandWrite(LCD_DISPLAY_ON_OFF | _displayfunction);
231 
232  // Display Control set
233  _displaycontrol = LCD_DISPLAY_SHIFT | LCD_SHIFT_LEFT;
234  LCDcommandWrite(LCD_MV_CUR_SHIFT_DISPLAY | _displaycontrol);
235 
236  // clear display and return cursor to home position. (Address 0)
237  LCDclear();
238 }

442 // read either command or data
443 static unsigned char LCDreceive(unsigned char RsMode) {
444  unsigned char highnib;
445  unsigned char lownib;
446 
447  LCDwritePCF8574(LCD_PCF8574_WEAK_PU | (En & ~En) | RsMode); // Set P7..P4 = 1, En = 0, RnW = 0, Rs = XX
448  highnib = LCDread4bits(LCD_PCF8574_WEAK_PU | En | RsMode);
449  lownib = LCDread4bits(LCD_PCF8574_WEAK_PU | En | RsMode);
450  LCDwritePCF8574((LCD_PCF8574_WEAK_PU & ~LCD_PCF8574_WEAK_PU) | En | RsMode); // Set P7..P4 = 1, En = 1, RnW = 0, Rs = XX
451  return (unsigned char) ((highnib & 0xF0) | ((lownib & 0xF0) >> 4));
452 }
